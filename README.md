```
    ├── Paragraph.php
    ├── PullQuote.php
    ├── StandFirst.php
    ├── SuperHero.php
    ├── Telemetry_acquisition.php
    └── views

        ├── footer.twig
        ├── gallery.twig
        ├── heading.twig
        ├── html.twig
        ├── image.twig
        ├── paragraph-social.twig
        ├── paragraph-template.twig
        ├── paragraph.twig
        ├── pull-quote.twig
        ├── standfirst.twig
        └── super-hero.twig
```
